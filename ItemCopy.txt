waitForKeyElements ("#search-results-0", addCopyLinks);

function addCopyLinks() {
    var rows = $('*[id^="item-container-"]');

    rows.each(function(){

        var that = $(this),
            id = $(this).attr('id').match(/[0-9]+/),
            link = $('<a>').attr('id', id[0]).addClass('copy2POB').text('Copy');

        link.click(function(){
            var poeObject = new PoeItem(that);
            var currentLink = $(this);

            var clipboard = new ClipboardJS('.copy2POB', {
                text: function(trigger) {
                    return poeObject.buildItem();
                }
            });

            clipboard.on('success', function(e) {
                $(currentLink).text('Copied to clipboard');
            });

        });

        $(this).find(".bottom-row .proplist").append($('<li>').html(link));
    });
}

    this.staves = "Iron Staff, Coiled Staff, Vile Staff, Military Staff, Serpentine Staff, Foul Staff, Ezomyte Staff, Maelstr\u00f6m Staff, Judgement Staff, Gnarled Branch, Primitive Staff, Long Staff, Royal Staff, Crescent Staff, Woodful Staff, Quarterstaff, Highborn Staff, Moon Staff, Primordial Staff, Lathi, Imperial Staff, Eclipse Staff";

    this.item_rarities = ["Normal", "Magic", "Rare", "Unique", "Gem", "Currency", "", "", "", "Relic"];

    this.returnWeaponRange = function() {

         if (this.runeDaggers.indexOf(this.itemType) !== -1)
            return 10;

        if (this.thrustSwords.indexOf(this.itemType) !== -1)
            return 14;

        if (this.staves.indexOf(this.itemType) !== -1)
            return 13;

        if (this.itemBase == "Dagger")
            return 10;

        if (this.itemBase == "Sceptre")
            return 11;

        if (this.itemBase == "Claw")
            return 11;

        if (this.itemBase == "One Hand Mace")
            return 11;

        if (this.itemBase == "Fishing Rods")
            return 13;

        if (this.itemBase.indexOf("One ") !== -1)
            return 11;

        if (this.itemBase.indexOf("Two ") !== -1)
            return 13;

        return -1;

    };


    this.returnItemQuality = function() {

        if (this.doNotIncludeQuality.indexOf(this.itemBase) === -1)
        {
            //need to check for  Quality (Attribute Modifiers) items
            var implicitQuality = this.returnQualityMods();
            var quality = this.htmlDiv.find("td[data-name='q']").data("value");

            if (implicitQuality === "")
            {
                if (quality !== "" && quality !== 0)
                {
                    quality = "Quality: +" + quality + "%";
                }

            } else {
                quality = implicitQuality;
            }


            return quality;
        }

        return "";
    };

    this.buildItemNameAndType = function() {

        var main = this;

        var fullTitle = this.htmlDiv.find(".title").text().trim();

        var isUnique = this.returnRarity() === "Unique";

        if (fullTitle.indexOf("Synthesised") !== -1)
            this.isSynthesised = true;

        if (fullTitle.indexOf("corrupted") !== -1)
            this.isCorrupted = true;

        fullTitle = fullTitle.replace(" Synthesised", "");
        fullTitle = fullTitle.replace("corrupted ", "");

        var found = false;

        for (var tempBase in this.items_types) {
            var commaSplitList = String(this.items_types[tempBase]).split(',');
            for (var i = 0; i < commaSplitList.length; i++) {

                if (fullTitle.indexOf(commaSplitList[i]) !== -1 && commaSplitList[i] !== '') {

                    if (fullTitle.endsWith(commaSplitList[i])) {
                        fullTitle = fullTitle.replace(commaSplitList[i], "");
                    }

                    if (isUnique === true && main.isSynthesised === true)
                        commaSplitList[i] = "Synthesised " + commaSplitList[i];

                    this.itemType = commaSplitList[i];
                    this.itemName = fullTitle;

                    found = true;

                    break;
                }
            }

            if (found == true) {
                this.itemBase = tempBase;
                break;
            }
        }

        if (found == false) {
            this.itemName = fullTitle;
            this.itemBase = 'Unknown';
            this.itemType = 'Unknown';
        }

    };

    this.returnItemSockets = function() {
        var sockets = "";
        var find = this.htmlDiv.find(".sockets .sockets-inner div");
        for (var i = 0; i < find.length; i++) {
            var a = find[i];
            var classes = a.className.split(' ');
            if (classes.indexOf('socketLink') >= 0) {
                sockets += "-";
            } else {
                if (this.regex.exec(sockets)) {
                    sockets += " ";
                }
                if (classes.indexOf('socketI') >= 0) {
                    sockets += "B";
                } else if (classes.indexOf('socketD') >= 0) {
                    sockets += "G";
                } else if (classes.indexOf('socketS') >= 0) {
                    sockets += "R";
                } else if (classes.indexOf('socketG') >= 0) {
                    sockets += "W";
                }
            }
        }
        return sockets;
    };

    this.returnWeaponStats = function() {

        var physDamage = this.htmlDiv.find("td[data-name='quality_pd']").text().trim();
        var eleDamage = this.htmlDiv.find("td[data-name='ed']").text().trim();
        var critChance = this.htmlDiv.find("td[data-name='crit']").text().trim();
        var attacksPerSec = this.htmlDiv.find("td[data-name='aps']").text().trim();

        var stats = "";

        if (physDamage !== "")
            stats += "Physical Damage: " + physDamage + "\r\n";

        if (eleDamage !== "")
            stats += "Elemental Damage: " + eleDamage + "\r\n";

        if (critChance !== "")
            //use 2 decimal places
            stats += "Critical Strike Chance: " + (Math.round(critChance * 100) / 100).toFixed(2) + "%\r\n";

        if (attacksPerSec !== "")
            stats += "Attacks per Second: " + attacksPerSec + "\r\n";

        var range = this.returnWeaponRange();
        if (range !== -1)
            stats += "Weapon Range: " + range + "\r\n";

        return stats.trim();

    };

    this.returnRequirementStats = function() {

        var stats = "";

        this.htmlDiv.find("ul.requirements.proplist li").each(function(){
            var that = $(this);

            if (that.text().indexOf('Strength') !== -1)
                stats += "Str: " + that.text().replace("Strength: ","").trim() + '\r\n';

            if (that.text().indexOf('Dexterity') !== -1)
                stats += "Dex: " + that.text().replace("Dexterity: ","").trim() + '\r\n';

            if (that.text().indexOf('Intelligence') !== -1)
                stats += "Int: " + that.text().replace("Intelligence: ","").trim() + '\r\n';
        });

        return stats.trim();

    };

    this.returnArmourStats = function() {

        var block = this.htmlDiv.find("td[data-name='block']").text().trim();
        var armour = this.htmlDiv.find("td[data-name='quality_armour']").text().trim();
        var evasion = this.htmlDiv.find("td[data-name='quality_evasion']").text().trim();
        var energyShield = this.htmlDiv.find("td[data-name='quality_shield']").text().trim();

        var stats = "";

        if (block !== "")
            stats += "Chance to Block: " + Math.round(block) + "%\r\n";

        if (armour !== "")
            stats += "Armour: " + Math.round(armour) + "\r\n";

        if (evasion !== "")
            stats += "Evasion Rating: " + Math.round(evasion) + "\r\n";

        if (energyShield !== "")
            stats += "Energy Shield: " + Math.round(energyShield) + "\r\n";

        return stats.trim();

    };

    this.returnLevelRequirement = function() {
        var level = this.htmlDiv.find(".first-line li").first().text().replace("Level:", "").replace("ilvl:","").trim();
        if (level == "")
            level = "1";
        return level;
    };

    this.returnItemLevel = function() {
        var level = this.htmlDiv.find("span[data-name='ilvl']").text().replace("ilvl:", "").trim();
        if (level == "")
            level = "1";
        return level;
    };

    this.returnIsFracturered = function() {
        return this.htmlDiv.find('.label-fractured').text() !== ""
    };

    this.returnRarity = function() {
        var rarityIndex = this.htmlDiv.find(".title").attr("class").match(/[0-9]/);
        return this.item_rarities[Number(rarityIndex)];
    };

    this.returnImplicitMods = function() {
        var itemMods = "";

        this.htmlDiv.find(".withline li").each(function(){
            var that = $(this);
            itemMods += that.text() + '\r\n';
        });
        return itemMods.trim();
    };

    this.returnQualityMods = function() {
        var itemMods = "";

        //this will find items like Quality (Attribute Modifiers): +16%
        this.htmlDiv.find("ul.item-mods li:not([class])").each(function(){
            var that = $(this);
            if (that.text().indexOf('Quality (') !== -1) {
                itemMods += that.text() + '\r\n';
            }
        });

        return itemMods.trim();
    };

    this.returnEnchantedMods = function() {
        var itemMods = "";
        this.htmlDiv.find("ul .mods:not(.withline) li").each(function(){
            var that = $(this);

            if (that.text().indexOf('enchanted') !== -1) {
                itemMods += that.text().replace("enchanted", "").trim() + '\r\n';
            }

        });
        return itemMods.trim();
    };

    this.returnCraftedMods = function() {
        var craftedItems = "";
        this.htmlDiv.find("ul .mods:not(.withline) li").each(function(){
            var that = $(this);

            if (that.text().indexOf('crafted') !== -1) {
                craftedItems += that.text().replace("crafted", "").trim() + '\r\n';
            }
        });
        return craftedItems.trim();
    };

    this.returnItemMods = function() {
        var itemMods = "";
        var main = this;
        this.htmlDiv.find("ul .mods:not(.withline) li").each(function(){
            var that = $(this);
            if (that.text().indexOf('enchanted') === -1 &&
                that.text().indexOf('crafted') === -1 &&
                that.text().indexOf('total') === -1 &&
                that.text().indexOf('pseudo') === -1) {

                var item = that.contents().not($("span")).text().trim();
                if (item.indexOf('Crusader') !== -1)
                     main.isCrusader = true;
                else if (item.indexOf('Warlord') !== -1)
                     main.isWarlord = true;
                else if (item.indexOf('Redeemer') !== -1)
                     main.isRedeemer = true;
                else if (item.indexOf('Hunter') !== -1)
                     main.isHunter = true;
                else if (item.indexOf('Shaper') !== -1)
                     main.isShaper = true;
                else if (item.indexOf('Elder') !== -1)
                     main.isElder = true;
                else if (item.indexOf('Mirrored') !== -1)
                    main.isMirror = true;
                else
                    itemMods += item + '\r\n';

            }
        });
        return itemMods.trim();
    };

}

PoeItem.prototype.buildItem = function() {

    this.buildItemNameAndType();

    var item = "";

    var rarity = this.returnRarity();
    if (rarity !== "")
        item += "Rarity: " + rarity + '\r\n';

    if (this.itemName != "")
        item += this.itemName + '\r\n';
    item += this.itemType + '\r\n';
    item += '--------' + '\r\n';

    if (this.includeBaseType.indexOf(this.itemBase) !== -1)
    {
        item += this.itemBase + '\r\n';
        item += this.returnWeaponStats() + '\r\n';
        item += '--------' + '\r\n';
    }

    var quality = this.returnItemQuality();
    if (quality !== "")
        item += quality + '\r\n';

    var armourStats = this.returnArmourStats();
    if (armourStats !== "")
        item += armourStats + '\r\n';

    if (armourStats !== "" || quality !== "")
        item += '--------' + '\r\n';

    item += 'Requirements: ' + '\r\n';
    item += 'Level: ' + this.returnLevelRequirement() + '\r\n';
    var requirementStats = this.returnRequirementStats();
    if (requirementStats !== "")
        item += this.returnRequirementStats() + '\r\n';
    item += '--------' + '\r\n';

    var sockets = this.returnItemSockets();
    if (sockets !== "")
    {
        item += "Sockets: " + sockets + '\r\n';
        item += '--------' + '\r\n';
    }

    item += 'Item Level: ' + this.returnItemLevel() + '\r\n';
    item += '--------' + '\r\n';

    var enchanted = this.returnEnchantedMods();
    if (enchanted !== "")
    {
        item += enchanted + '\r\n';
        item += '--------' + '\r\n';
    }

    var implicit = this.returnImplicitMods();
    if (implicit !== "")
    {
        item += implicit + '\r\n';
        item += '--------' + '\r\n';
    }

    item += this.returnItemMods() + '\r\n';

    var crafted = this.returnCraftedMods();
    if (crafted !== "")
        item += crafted + '\r\n';

    item += '--------' + '\r\n';

    if (this.isCorrupted == true)
    {
        item += 'Corrupted' + '\r\n';
        item += '--------' + '\r\n';
    }

    if (this.isCrusader == true)
        item += 'Crusader Item' + '\r\n';
    if (this.isWarlord == true)
        item += 'Warlord Item' + '\r\n';
    if (this.isRedeemer == true)
        item += 'Redeemer Item' + '\r\n';
    if (this.isHunter == true)
        item += 'Hunter Item' + '\r\n';
    if (this.isElder == true)
        item += 'Elder Item' + '\r\n';
    if (this.isShaper == true)
        item += 'Shaper Item' + '\r\n';


    if (this.isCrusader == true || this.isWarlord == true || this.isRedeemer == true || this.isHunter == true || this.isElder == true || this.isShaper == true)
        item += '--------' + '\r\n';

    if (this.returnIsFracturered() == true)
    {
        item += 'Fractured Item' + '\r\n';
        item += '--------' + '\r\n';
    }

    if (this.isSynthesised == true)
    {
        item += 'Synthesised Item' + '\r\n';
        item += '--------' + '\r\n';
    }

    if (this.isMirror == true)
    {
        item += 'Mirrored' + '\r\n';
        item += '--------' + '\r\n';
    }

    //remove bottom ------ if it exists
    if (item.charAt(item.length - 3) == '-') {
        item = item.trim();
        item = item.substr(0, item.length - 8);
    }

    return item.trim();

};
